   <ul class="tabNav">
      <li (click)="onChangeTab()" [class.active]="!tab"><a>Tab 1</a></li>

      <li (click)="onChangeTab('tab3')" [class.active]="tab === 'tab3'"><a>Tab 3</a></li>
